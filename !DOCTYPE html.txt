<script>
  const totalSeats = 10;
  const occupiedSeats = [1,2,3,4,5,6,7];
  let selectedSeat = null;

  const seatsDiv = document.getElementById("seats");

  function loadSeats() {
    for (let i = 1; i <= totalSeats; i++) {
      const seat = document.createElement("div");
      seat.classList.add("seat");
      seat.textContent = i;

      if (occupiedSeats.includes(i)) {
        seat.classList.add("occupied");
      } else {
        seat.classList.add("available");
        seat.onclick = () => selectSeat(i);
      }

      seatsDiv.appendChild(seat);
    }
  }

  function selectSeat(num) {
    selectedSeat = num;
    document.getElementById("seat-num").textContent = num;
    document.getElementById("booking-box").style.display = "block";
  }

  function confirmBooking() {
    const name = document.getElementById("name").value.trim();
    const phone = document.getElementById("phone").value.trim();
    
    if (!name || !phone) {
      alert("Please enter your name and phone number");
      return;
    }

    alert("Seat " + selectedSeat + " booked successfully!\nName: " + name + "\nPhone: " + phone);
  }

  loadSeats();
</script>
